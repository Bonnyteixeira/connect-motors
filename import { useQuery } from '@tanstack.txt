import { useQuery } from '@tanstack/react-query';
import api from '../lib/api';

export function useFetch<T>(key: string, endpoint: string) {
  return useQuery<T, Error>({
    queryKey: [key],
    queryFn: async () => {
      const { data } = await api.get<T>(endpoint);
      return data;
    }
  });
}
